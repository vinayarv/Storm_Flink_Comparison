CREATE TABLE T_FLINK_PCT_KAFKA
AS
   SELECT COUNT REC_EMIT,
          COUNT (1) REC_PRCSD,
          (COUNT (1) / COUNT) * 100 PCT_PRCSD,
          WIND_TS
     FROM (SELECT COUNT (1) COUNT, wind_ts
             FROM (SELECT READ_TS,
                          WRITE_TS,
                          WRITE_TS - READ_TS latency,
                          wind_ts
                     FROM (SELECT FLINK.*, wind_ts
                             FROM (SELECT *
                                     FROM (SELECT READ_TS + INTERVAL Seq_no SECOND wind_ts
    FROM
        (SELECT
        MIN(READ_TS) read_ts
    FROM
        T_FLINK_BENCHMARK) MIN_READ_TS, (SELECT
        @rownum:=@rownum + 1 AS Seq_no
    FROM
        (SELECT @rownum:=0) SEQ, (SELECT
        1
        FROM
        T_FLINK_BENCHMARK) DUMMY_REC) SEQ_NUM_GEN) WIN_SLOTS  LIMIT 35) g LEFT JOIN T_FLINK_BENCHMARK FLINK
    ON (
        READ_TS  >= (wind_ts - INTERVAL 10 SECOND)
            AND READ_TS  <= wind_ts)) WIN_READ_REC) h
GROUP BY wind_ts) WIN_READ_REC  LEFT JOIN T_FLINK_BENCHMARK FLINK
ON ( FLINK.write_ts >= (WIN_READ_REC.wind_ts - INTERVAL 10 SECOND) 
    and FLINK.write_ts <= WIN_READ_REC.wind_ts 
    AND FLINK.read_ts >=  (WIN_READ_REC.wind_ts - INTERVAL 10 SECOND)  
    AND FLINK.read_ts <= WIN_READ_REC.WIND_TS)
GROUP BY WIND_TS;
